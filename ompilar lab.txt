 for(int i = 0; i<n/2; i++){

int main(){
    
    
   char str[100];
   int vowelCounter = 0, consonantCounter = 0;
    cout<<endl<<endl;
   cout << "Enter any string: ";
   cin.getline(str, 150);
   
   
   for(int i = 0; str[i]!='\0'; i++) {
      if(str[i]=='a' || str[i]=='e' || str[i]=='i' ||
         str[i]=='o' || str[i]=='u' || str[i]=='A' ||
         str[i]=='E' || str[i]=='I' || str[i]=='O' ||
         str[i]=='U')

         vowelCounter++;
